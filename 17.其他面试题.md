10. 实现一个immutable

```js
function immutable(obj) {
  return {
    set(path, val) {
      const paths = path.split('.')
      const lastK = paths.pop()
      this.get(paths.join('.'))[lastK] = val
      return this
    },
    get(path) {
      const paths = path.split('.')
      return paths.reduce((a, k) => a[k], obj)
    }
  }
}
```
